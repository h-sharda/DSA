// Problem: 1015. Smallest Integer Divisible by K (Medium)
// Link: https://leetcode.com/problems/smallest-integer-divisible-by-k

class Solution {
public:
    int smallestRepunitDivByK(int k) {
        if (k % 2 == 0 || k % 5 == 0) return -1;

        int rem = 1 % k, len = 1;

        while (rem != 0) {
            rem = (rem * 10 + 1) % k;
            len++;
        }

        return len;
    }
};
